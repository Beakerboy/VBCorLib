@remarks If a Byte array is passed in, then the bytes from the underlying stream are copied to the array. If an Integer array is passed in, then enough bytes are read from the stream to produce the requested number of decoded characters. The decoded characters are placed in the array starting at Index.

The position within the stream is advanced based on the Encoding being used.

If all parameters are missing then a single character will be decoded from the stream and returned.

All parameters must either be missing or supplied or an exception is thrown.

Private Sub ReadExample()
    Dim Bytes(0 To 1) As Byte
    Dim Chars(0 To 1) As Integer
    Dim Reader As BinaryReader
    
    Set Reader = NewBinaryReader(NewMemoryStream(NewBytes(49, 50, 51, 194, 163, 54), Mode:=CopyMode))
    
    Console.WriteLine "Reads next character"
    Console.WriteLine "Next character: {0}({0:$})", Reader.Read
    Console.WriteLine
    
    Console.WriteLine "Read next two bytes"
    Console.WriteLine "Bytes read: {0}", Reader.Read(Bytes, 0, 2)
    Console.WriteLine "Bytes: {0}, {1}", Bytes(0), Bytes(1)
    Console.WriteLine
    
    Console.WriteLine "Read next two characters"
    Console.WriteLine "Characters read: {0}", Reader.Read(Chars, 0, 2)
    Console.WriteLine "Characters: {0}({0:$}), {1:$}", Chars(0), Chars(1)
    
    Console.ReadKey

    ' outputs:
    ' Reads next character
    ' Next character: 49(1)
    '
    ' Read next two bytes
    ' Bytes read: 2
    ' Bytes: 50, 51
    '
    ' Read next two characters
    ' Characters read: 2
    ' Characters: 163(Â£), 6
End Sub
