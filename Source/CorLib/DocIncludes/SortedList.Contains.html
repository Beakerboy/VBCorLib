@example The following code example shows how to determine whether a SortedList object contains a specific element. 
Public Sub Main()
    Dim List As New SortedList
    Dim KeyToFind As Integer
    Dim ValueToFind As String
    
    List.Add 2, "two"
    List.Add 4, "four"
    List.Add 1, "one"
    List.Add 3, "three"
    List.Add 0, "zero"
    
    PrintIndexAndKeysAndValues List
    
    '' Searched for a specific key.
    KeyToFind = 2
    Debug.Print CorString.Format("The key ""{0}"" is {1}.", KeyToFind, IIf(List.ContainsKey(KeyToFind), "in the SortedList", "NOT in the SortedList"))
    KeyToFind = 6
    Debug.Print CorString.Format("The key ""{0}"" is {1}.", KeyToFind, IIf(List.ContainsKey(KeyToFind), "in the SortedList", "NOT in the SortedList"))
    
    '' Searches for a specific value.
    ValueToFind = "three"
    Debug.Print CorString.Format("The value ""{0}"" is {1}.", ValueToFind, IIf(List.ContainsValue(ValueToFind), "in the SortedList", "NOT in the SortedList"))
    ValueToFind = "nine"
    Debug.Print CorString.Format("The value ""{0}"" is {1}.", ValueToFind, IIf(List.ContainsValue(ValueToFind), "in the SortedList", "NOT in the SortedList"))
End Sub

Private Sub PrintIndexAndKeysAndValues(ByVal List As SortedList)
    Dim i As Long
    
    Debug.Print "Index", "Key", "Value"
    
    For i = 0 To List.Count - 1
        Debug.Print i, List.GetKey(i), List.GetByIndex(i)
    Next
    
    Debug.Print
End Sub

'' This code produces the following output.
''
''    Index         Key           Value
''     0             0            zero
''     1             1            one
''     2             2            two
''     3             3            three
''     4             4            four
''
''    The key "2" is in the SortedList.
''    The key "6" is NOT in the SortedList.
''    The value "three" is in the SortedList.
''    The value "nine" is NOT in the SortedList.
